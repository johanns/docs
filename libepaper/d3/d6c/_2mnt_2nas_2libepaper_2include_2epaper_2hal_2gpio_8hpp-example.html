#pragma once
#include <concepts>
namespace epaper::hal {
template <typename T>
concept DigitalOutput = requires(T out, bool level) {
{ out.write(level) } -> std::same_as<void>;
};
template <typename T>
concept DigitalInput = requires(T in) {
{ in.read() } -> std::same_as<bool>;
};
} // namespace epaper::hal
