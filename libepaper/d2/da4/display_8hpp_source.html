348 // ========== Drawing Operations (Builder API) ==========
349 //
350 // Modern fluent builder interface for drawing operations.
351 // All drawing operations:
352 // - Out-of-bounds coordinates are silently clipped (no errors)
353 // - Operations never fail or throw exceptions
354 // - Coordinates are in logical space (accounting for display orientation)
355 // - Drawing is performed on the framebuffer; call refresh() to update the display
356 //
357 // Use the factory methods (line(), rectangle(), etc.) to create builders,
358 // then call build() to produce a command, and pass it to draw().
359 //
360 // @example
361 // @code{.cpp}
362 // display.draw(
363 // display.line()
364 // .from({10, 10})
365 // .to({100, 100})
366 // .color(Color::Black)
367 // .width(DotPixel::Pixel2x2)
368 // .build()

733[[nodiscard]] auto create_display(Device &device, DisplayMode mode, Orientation orientation = Orientation::Portrait0,
734 bool auto_sleep = true) -> std::expected<Display, Error> {
735 auto driver = std::make_unique<DriverType>(device);
736 if (auto result = driver->init(mode); !result) {
737 return std::unexpected(result.error());
738 }
739 return Display(std::move(driver), orientation, auto_sleep);
