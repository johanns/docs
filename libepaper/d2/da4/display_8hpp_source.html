596template <typename DriverType>
597[[nodiscard]] auto create_display(Device &device, DisplayMode mode, Orientation orientation = Orientation::Portrait0,
598 bool auto_sleep = true) -> std::expected<Display, Error> {
599 auto driver = std::make_unique<DriverType>(device);
600 if (auto result = driver->init(mode); !result) {
601 return std::unexpected(result.error());
602 }
603 return Display(std::move(driver), orientation, auto_sleep);
604}
